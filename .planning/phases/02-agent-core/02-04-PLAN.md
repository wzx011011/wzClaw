       - Change `registerIpcHandlers(gateway: LLMGateway)` to `registerIpcHandlers(gateway: LLMGateway, agentLoop: AgentLoop, permissionManager: PermissionManager)`
       - Keep the existing settings handlers (`settings:get`, `settings:update`) exactly as they are — do not modify them.
       - Replace the `agent:send_message` handler:
         ```
         ipcMain.handle('agent:send_message', async (event, request) => {
           // Validate with Zod (keep existing validation)
           const result = IpcSchemas['agent:send_message'].request.safeParse(request)
           if (!result.success) throw new Error(`Invalid request: ${result.error.message}`)

           const sender = event.sender

           // Build AgentConfig from current settings
           const config: AgentConfig = {
             model: currentSettings.model,
             systemPrompt: currentSettings.systemPrompt ?? DEFAULT_SYSTEM_PROMPT,
             workingDirectory: process.cwd(), // Phase 3 adds workspace selection
             conversationId: result.data.conversationId,
           }

           // Run agent loop and forward events to renderer
           try {
             for await (const agentEvent of agentLoop.run(result.data.content, config, sender)) {
               switch (agentEvent.type) {
                 case 'agent:text':
                   sender.send('stream:text_delta', { content: agentEvent.content })
                   break
                 case 'agent:tool_call':
                   sender.send('stream:tool_use_start', { id: agentEvent.toolCallId, name: agentEvent.toolName })
                   break
                 case 'agent:tool_result':
                   sender.send('stream:tool_use_end', { id: agentEvent.toolCallId, parsedInput: { output: agentEvent.output, isError: agentEvent.isError, toolName: agentEvent.toolName } })
                   break
                 case 'agent:permission_request':
                   sender.send('agent:permission_request', { toolName: agentEvent.toolName, toolInput: agentEvent.input, reason: 'This tool can modify your files. Approve?' })
                   break
                 case 'agent:error':
                   sender.send('stream:error', { error: agentEvent.error })
                   break
                 case 'agent:done':
                   sender.send('stream:done', { usage: agentEvent.usage })
                   break
               }
             }
           } catch (error) {
             sender.send('stream:error', { error: error instanceof Error ? error.message : String(error) })
           }
         })
         ```
       - Replace the `agent:stop` handler:
         ```
         ipcMain.handle('agent:stop', () => {
           agentLoop.cancel()
         })
         ```
       - Add `agent:permission_response` handler — this is a temporary handler registration pattern. The PermissionManager will register one-shot handlers when it needs a response. So just add a note/comment here. The actual handler is managed by PermissionManager.
       - Add cleanup: on `sender` destroyed (window closed), call `agentLoop.cancel()` and `permissionManager.clearSession(conversationId)`.
