       export interface AgentTask {

         status: TaskStatus
         blockedBy: string[]    // task IDs this task depends on

       }

    2. Create `src/main/tasks/task-manager.ts`:
       - Export interface `TaskEvent`:
         ```typescript
         export type TaskEvent =
           | { type: 'task:created'; task: AgentTask }
           | { type: 'task:updated'; task: AgentTask }
         ```
       - Export class `TaskManager`:
         - Private: `tasks: Map<string, AgentTask>`, `listeners: Set<(event: TaskEvent) => void>`, `nextId: number = 1`
         - `createTask(subject: string, description: string, blockedBy: string[] = []): AgentTask`:
           - Generate id: `task-${this.nextId++}`
           - Check circular dependencies: validate that none of the blockedBy task IDs transitively depend on this new task. Simple DFS: for each blocker, check if it has a blockedBy that includes this task's would-be blockers. Since we haven't created the task yet, just check that blockedBy IDs exist in the tasks map (or skip validation if they don't exist yet -- allow forward references).
           - Determine initial status: if any blockedBy task is not completed, set status to 'blocked'; otherwise 'pending'
           - Create AgentTask with timestamps
           - Store in map
           - Emit task:created event
           - Return the task
         - `updateTask(taskId: string, updates: Partial<Pick<AgentTask, 'status' | 'subject' | 'description' | 'blockedBy'>>): AgentTask | null`:
           - Get task from map, return null if not found
           - Apply updates
           - If status changed to 'completed', check all other tasks that are blocked by this task -- if all their blockers are now completed, set their status to 'pending' and emit task:updated for each
           - If blockedBy updated, re-evaluate status
           - Update updatedAt timestamp
           - Emit task:updated event
           - Return updated task
         - `getTask(id: string): AgentTask | undefined`
         - `getAllTasks(): AgentTask[]`
         - `clearTasks(): void` -- clears all tasks (for new session)
         - `onTaskEvent(listener: (event: TaskEvent) => void): () => void` -- subscribe, returns unsubscribe
         - Private `emit(event: TaskEvent): void` -- call all listeners
